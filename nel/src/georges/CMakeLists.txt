FILE(GLOB SRC *.cpp)
FILE(GLOB PRIV_H *.h)
FILE(GLOB PUB_H ../../include/nel/georges/*.h)

SOURCE_GROUP(headers FILES ${PRIV_H} ${PUB_H})

DECORATE_NEL_LIB("nelgeorges")
SET(NLGEORGES_LIB ${LIBNAME})
DECORATE_NEL_LIB("nelmisc")
SET(NLMISC_LIB ${LIBNAME})


IF(NOT WIN32)
  ADD_LIBRARY(${NLGEORGES_LIB} SHARED ${SRC})
ELSE(NOT WIN32)
  ADD_LIBRARY(${NLGEORGES_LIB} STATIC ${SRC})
ENDIF(NOT WIN32)

INCLUDE_DIRECTORIES(${LIBXML2_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(${NLGEORGES_LIB} ${NLMISC_LIB})
SET_TARGET_PROPERTIES(${NLGEORGES_LIB} PROPERTIES VERSION ${NL_VERSION})

IF(WIN32)
  SET_TARGET_PROPERTIES(${NLGEORGES_LIB} PROPERTIES COMPILE_FLAGS "/Yustdgeorges.h")
  SET_SOURCE_FILES_PROPERTIES(stdgeorges.cpp PROPERTIES COMPILE_FLAGS "/Ycstdgeorges.h")
  SET_TARGET_PROPERTIES(${NLGEORGES_LIB} PROPERTIES LINK_FLAGS "/NODEFAULTLIB:libcmt")
ENDIF(WIN32)

ADD_DEFINITIONS(${LIBXML2_DEFINITIONS})

INSTALL(TARGETS ${NLGEORGES_LIB} LIBRARY DESTINATION lib ARCHIVE DESTINATION lib)
