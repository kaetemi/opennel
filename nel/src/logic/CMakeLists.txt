FILE(GLOB SRC *.cpp *.h)

DECORATE_NEL_LIB("nellogic")
SET(NLLOGIC_LIB ${LIBNAME})
DECORATE_NEL_LIB("nelmisc")
SET(NLMISC_LIB ${LIBNAME})

IF(NOT WIN32)
  ADD_LIBRARY(${NLLOGIC_LIB} SHARED ${SRC})
ELSE(NOT WIN32)
  ADD_LIBRARY(${NLLOGIC_LIB} STATIC ${SRC})
ENDIF(NOT WIN32)

INCLUDE_DIRECTORIES(${LIBXML2_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(${NLLOGIC_LIB} ${LIBXML2_LIBRARIES} ${NLMISC_LIB})
SET_TARGET_PROPERTIES(${NLLOGIC_LIB} PROPERTIES VERSION ${NL_VERSION})

IF(WIN32)
  SET_TARGET_PROPERTIES(${LIBNAME} PROPERTIES LINK_FLAGS "/NODEFAULTLIB:libcmt")
ENDIF(WIN32)


ADD_DEFINITIONS(${LIBXML2_DEFINITIONS})

INSTALL(TARGETS ${NLLOGIC_LIB} LIBRARY DESTINATION lib ARCHIVE DESTINATION lib)
