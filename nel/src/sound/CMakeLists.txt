FILE(GLOB SRC *.cpp *.h)

DECORATE_NEL_LIB("nelsound")
SET(NLSOUND_LIB ${LIBNAME})
DECORATE_NEL_LIB("nelligo")
SET(NLLIGO_LIB ${LIBNAME})
DECORATE_NEL_LIB("nelgeorges")
SET(NLGEORGES_LIB ${LIBNAME})
DECORATE_NEL_LIB("nel3d")
SET(NL3D_LIB ${LIBNAME})

IF(NOT WIN32)
  ADD_LIBRARY(${NLSOUND_LIB} SHARED ${SRC})
ELSE(NOT WIN32)
  ADD_LIBRARY(${NLSOUND_LIB} STATIC ${SRC})
ENDIF(NOT WIN32) 

INCLUDE_DIRECTORIES(${LIBXML2_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(${NLSOUND_LIB} ${LIBXML2_LIBRARIES} ${NLLIGO_LIB} ${NLGEORGES_LIB} ${NL3D_LIB})
SET_TARGET_PROPERTIES(${NLSOUND_LIB} PROPERTIES VERSION ${NL_VERSION})
ADD_DEFINITIONS(${LIBXML2_DEFINITIONS})

IF(WIN32)
  SET_TARGET_PROPERTIES(${NLSOUND_LIB} PROPERTIES COMPILE_FLAGS "/Yustdsound.h")
  SET_SOURCE_FILES_PROPERTIES(stdsound.cpp PROPERTIES COMPILE_FLAGS "/Ycstdsound.h")
  SET_TARGET_PROPERTIES(${NLSOUND_LIB} PROPERTIES LINK_FLAGS "/NODEFAULTLIB:libcmt")
ENDIF(WIN32)

INSTALL(TARGETS ${NLSOUND_LIB} LIBRARY DESTINATION lib ARCHIVE DESTINATION lib)

ADD_SUBDIRECTORY(driver)
