FILE(GLOB SRC *.cpp *.h)

DECORATE_NEL_LIB("nelsound")
SET(NLSOUND_LIB ${LIBNAME})
DECORATE_NEL_LIB("nelmisc")
SET(NLMISC_LIB ${LIBNAME})

ADD_LIBRARY(${NLSOUND_LIB} SHARED ${SRC})

INCLUDE_DIRECTORIES(${LIBXML2_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(${NLSOUND_LIB} ${LIBXML2_LIBRARIES} ${NLMISC_LIB})
SET_TARGET_PROPERTIES(${NLSOUND_LIB} PROPERTIES VERSION ${NL_VERSION})
ADD_DEFINITIONS(${LIBXML2_DEFINITIONS})

IF(WIN32)
  SET_TARGET_PROPERTIES(${NLSOUND_LIB} PROPERTIES COMPILE_FLAGS "/Yustdsound.h /FIstdsound.h")
  SET_TARGET_PROPERTIES(${NLSOUND_LIB} PROPERTIES LINK_FLAGS "/NODEFAULTLIB:libcmt")
ENDIF(WIN32)

INSTALL(TARGETS ${NLSOUND_LIB} LIBRARY DESTINATION lib)

SUBDIRS(driver)
