FILE(GLOB SRC *.cpp *.h)

DECORATE_NEL_LIB("nelpacs")
SET(NLPACS_LIB ${LIBNAME})
DECORATE_NEL_LIB("nelmisc")
SET(NLMISC_LIB ${LIBNAME})

IF(NOT WIN32)
  ADD_LIBRARY(${NLPACS_LIB} SHARED ${SRC})
ELSE(NOT WIN32)
  ADD_LIBRARY(${NLPACS_LIB} STATIC ${SRC})
ENDIF(NOT WIN32)

INCLUDE_DIRECTORIES(${LIBXML2_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(${NLPACS_LIB} ${LIBXML2_LIBRARIES} ${NLMISC_LIB})
SET_TARGET_PROPERTIES(${NLPACS_LIB} PROPERTIES VERSION ${NL_VERSION})
ADD_DEFINITIONS(${LIBXML2_DEFINITIONS})

INSTALL(TARGETS ${NLPACS_LIB} LIBRARY DESTINATION lib ARCHIVE DESTINATION lib)
