FILE(GLOB SRC *.cpp *.h)

DECORATE_NEL_LIB("nel3d")
SET(NL3D_LIB ${LIBNAME})
DECORATE_NEL_LIB("nelpacs")
SET(NLPACS_LIB ${LIBNAME})
DECORATE_NEL_LIB("nelig_lighter_lib")
SET(NLIG_LIGHTER_LIB ${LIBNAME})

IF(NOT WIN32)
  ADD_LIBRARY(${LIBNAME} SHARED ${SRC})
ELSE(NOT WIN32)
  ADD_LIBRARY(${LIBNAME} STATIC ${SRC})
ENDIF(NOT WIN32)

INCLUDE_DIRECTORIES(${LIBXML2_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(${NLIG_LIGHTER_LIB} ${LIBXML2_LIBRARIES} ${NL3D_LIB} ${NLPACS_LIB})
SET_TARGET_PROPERTIES(${NLIG_LIGHTER_LIB} PROPERTIES VERSION ${NL_VERSION})

IF(WIN32)
  SET_TARGET_PROPERTIES(${NLIG_LIGHTER_LIB} PROPERTIES LINK_FLAGS "/NODEFAULTLIB:libcmt")
ENDIF(WIN32)

ADD_DEFINITIONS(${LIBXML2_DEFINITIONS})

INSTALL(TARGETS ${NLIG_LIGHTER_LIB} LIBRARY DESTINATION lib ARCHIVE DESTINATION lib)
