FILE(GLOB SRC *.cpp *.h res/logic_editor.rc2)

ENABLE_LANGUAGE(RC)

DECORATE_NEL_LIB("logic_editor")
SET(LOGICEDITOR_LIB ${LIBNAME})
DECORATE_NEL_LIB("nellogic")
SET(NLLOGIC_LIB ${LIBNAME})
DECORATE_NEL_LIB("nelmisc")
SET(NLMISC_LIB ${LIBNAME})
DECORATE_NEL_LIB("nelnet")
SET(NLNET_LIB ${LIBNAME})

ADD_DEFINITIONS(-D_AFXDLL)
SET(CMAKE_MFC_FLAG 2) 
ADD_LIBRARY(${LOGICEDITOR_LIB} SHARED ${SRC})

INCLUDE_DIRECTORIES(${LIBXML2_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(${LOGICEDITOR_LIB} ${NLLOGIC_LIB} ${LIBXML2_LIBRARIES} ${NLMISC_LIB} ${NLNET_LIB})
SET_TARGET_PROPERTIES(${LOGICEDITOR_LIB} PROPERTIES VERSION ${NL_VERSION})

IF(WIN32)
  SET_TARGET_PROPERTIES(${LOGICEDITOR_LIB} PROPERTIES COMPILE_FLAGS "/Yustdafx.h")
  SET_SOURCE_FILES_PROPERTIES(stdafx.cpp PROPERTIES COMPILE_FLAGS "/Ycstdafx.h")
  SET_TARGET_PROPERTIES(${LOGICEDITOR_LIB} PROPERTIES LINK_FLAGS "/NODEFAULTLIB:libcmt")
ENDIF(WIN32)


ADD_DEFINITIONS(${LIBXML2_DEFINITIONS})

INSTALL(TARGETS ${LOGICEDITOR_LIB} LIBRARY DESTINATION lib ARCHIVE DESTINATION lib)
