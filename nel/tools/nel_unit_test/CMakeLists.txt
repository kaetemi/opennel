FILE(GLOB SRC *.cpp *.h)

DECORATE_NEL_LIB("nelmisc")
SET(NLMISC_LIB ${LIBNAME})

ADD_EXECUTABLE(nel_unit_test ${SRC})

INCLUDE_DIRECTORIES(${LIBXML2_INCLUDE_DIR} ${CPPTEST_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(nel_unit_test ${LIBXML2_LIBRARIES} ${CPPTEST_LIBRARY} ${PLATFORM_LINKFLAGS} ${NLMISC_LIB})
IF(WIN32)
  SET_TARGET_PROPERTIES(nel_unit_test PROPERTIES LINK_FLAGS "/NODEFAULTLIB:libcmt")
ENDIF(WIN32)
ADD_DEFINITIONS(${LIBXML2_DEFINITIONS})

INSTALL(TARGETS nel_unit_test RUNTIME DESTINATION bin)

ADD_SUBDIRECTORY(misc_ut)

IF(WITH_LIGO)
  ADD_SUBDIRECTORY(ligo_ut)
ENDIF(WITH_LIGO)

IF(WITH_NET)
  ADD_SUBDIRECTORY(net_ut)
ENDIF(WITH_NET)
