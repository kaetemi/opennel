FILE(GLOB SRC *.cpp *.h)

DECORATE_NEL_LIB("nelpacs")
SET(NLPACS_LIB ${LIBNAME})
DECORATE_NEL_LIB("nelligo")
SET(NLLIGO_LIB ${LIBNAME})
DECORATE_NEL_LIB("nel3d")
SET(NL3D_LIB ${LIBNAME})

ADD_EXECUTABLE(build_rbank ${SRC})

INCLUDE_DIRECTORIES(${LIBXML2_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(build_rbank ${LIBXML2_LIBRARIES} ${PLATFORM_LINKFLAGS} ${NLPACS_LIB} ${NL3D_LIB} ${NLLIGO_LIB})
IF(WIN32)
  SET_TARGET_PROPERTIES(build_rbank PROPERTIES LINK_FLAGS "/NODEFAULTLIB:libcmt")
ENDIF(WIN32)
ADD_DEFINITIONS(${LIBXML2_DEFINITIONS})

INSTALL(TARGETS build_rbank RUNTIME DESTINATION bin)
